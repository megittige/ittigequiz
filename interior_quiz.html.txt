        .results { display: none; margin-top: 20px; font-size: 18px; }

<body>
    <div class="quiz-container">
        <h1>Interior Design Style Quiz</h1>
        <div id="quiz"></div>
        <button id="submit" style="display:none; margin-top: 20px;">See Results</button>
        <div id="results" class="results"></div>
    </div>

    <script>
        const questions = [
            { text: "Choose Your Ideal Living Room Theme", images: ["livingroom1.jpg", "livingroom2.jpg"] },
            { text: "Pick Your Dream Kitchen Style", images: ["kitchen1.jpg", "kitchen2.jpg"] },
            { text: "What Kind of Bedroom Feels Most Like You?", images: ["bedroom1.jpg", "bedroom2.jpg"] },
            { text: "Select Your Preferred Flooring for Your Home", images: ["flooring1.jpg", "flooring2.jpg"] },
            { text: "Your Perfect Bathroom Would Have:", images: ["bathroom1.jpg", "bathroom2.jpg"] }
        ];

        let userChoices = [];
        let currentQuestion = 0;

        function loadQuestion() {
            if (currentQuestion >= questions.length) {
                document.getElementById("submit").style.display = "block";
                return;
            }
            const quizDiv = document.getElementById("quiz");
            quizDiv.innerHTML = `<div class='question'>${questions[currentQuestion].text}</div>`;
            questions[currentQuestion].images.forEach((img, index) => {
                quizDiv.innerHTML += `<img src="${img}" class='options' onclick='selectAnswer(${index})'>`;
            });
        }

        function selectAnswer(choice) {
            userChoices.push(choice);
            currentQuestion++;
            loadQuestion();
        }

        function showResults() {
            document.getElementById("quiz").style.display = "none";
            document.getElementById("submit").style.display = "none";
            const resultsDiv = document.getElementById("results");
            resultsDiv.style.display = "block";
            
            let style;
            const sum = userChoices.reduce((a, b) => a + b, 0);
            if (sum <= questions.length / 2) {
                style = "Modern Minimalist";
            } else {
                style = "Cozy Rustic";
            }
            
            resultsDiv.innerHTML = `<p>Your ideal interior design style is: <strong>${style}</strong></p>`;
        }

        document.getElementById("submit").addEventListener("click", showResults);
        loadQuestion();
    </script>
</body>
</html>
